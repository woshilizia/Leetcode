class Solution {
    public int mySqrt(int x) {
        if (x <= 1) return x;
        int left = 0, right = x;
        while (left <= right){
            int mid = (left+right)/2;
            if (x/mid>=mid && x/(mid+1)<(mid+1))
                return mid;
            else if (x/mid < mid)
                right = mid-1;
            else
                left = mid+1;
        }
        return left;
    }
}
